## qplot is great and provides useful defaults for quick graphs, but for more control you can use `ggplot`:

## The gg is for "Grammar of Graphics" (Wilkinson 2005). 
## This consists of layers you chain together using the `+` operator

# 1. The data itself
# 2. Aesthetic mappings (e.g. length -> x axis)
# 2. Geometric objects (e.g. histogram, scatterplot, lines, text etc.)
# 3. Statistical transformations (e.g. best fit lines, binning, counts of observations)
# 4. Scales (legends and axes)
# 5. Coordinate system (generally Cartesian but can be changed for maps etc)
# 6. facets to break into subplots

## It is important to think about the structure of your data when you are building a visualisation - reshaping your data is a key skill


## let's bootstrap a model:

# What does this function do?
boot_lm <- function(formula, data, ...){
  function(){
    lm(formula = formula,
       data = data[sample(nrow(data), replace=TRUE),], ...)
  }
}

bristles_boot <- boot_lm(bristles ~ length + width + sex + treatment, dat)

bristles_boot() # New bootstrapped model every time it is called

bstrap <- data.frame(t(sapply(X=1:1000,
                 FUN = function(x) bristles_boot()$coef)))
